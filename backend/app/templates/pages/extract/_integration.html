Code Examples

cURL

curl -X POST "{{ request.url.scheme }}://{{ request.url.netloc }}/api/v1/extract" \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: multipart/form-data" \
  -F "file=@document.pdf" \
  -F "template_id=YOUR_TEMPLATE_ID"

Python

import requests

url = "{{ request.url.scheme }}://{{ request.url.netloc }}/api/v1/extract"
headers = {"Authorization": "Bearer YOUR_API_KEY"}

with open("document.pdf", "rb") as f:
    response = requests.post(
        url,
        headers=headers,
        files={"file": f},
        data={"template_id": "YOUR_TEMPLATE_ID"}
    )

result = response.json()
print(result["extracted_data"])

JavaScript

const formData = new FormData();
formData.append('file', document.querySelector('input[type="file"]').files[0]);
formData.append('template_id', 'YOUR_TEMPLATE_ID');

const response = await fetch('{{ request.url.scheme }}://{{ request.url.netloc }}/api/v1/extract', {
  method: 'POST',
  headers: {
    'Authorization': 'Bearer YOUR_API_KEY'
  },
  body: formData
});

const result = await response.json();
console.log(result.extracted_data);
